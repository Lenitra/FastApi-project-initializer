Satellite                       # Informations générales sur un satellite
- name str .max50 .unique .nn                  # Nom du satellite (ex: ISS)
- object_id str .max20 .unique .nn             # ID standard international (ex: 1998-067A)
- norad_cat_id int .unique .nn                 # Identifiant NORAD (ex: 25544)
- object_type str .max20                       # Type d’objet (ex: PAYLOAD, DEBRIS)
- classification str .max5                     # Niveau de classification (U = Unclassified)
- rcs_size str .max15                          # Taille estimée de la surface radar (ex: LARGE)
- country str .max10                           # Code pays ou entité opératrice (ex: USA, RUS)
- launch_date date .past .nn                   # Date de lancement (dans le passé)
- launch_site str .max20                       # Lieu de lancement (ex: TTMTR)
- decay_date date .past                        # Date de rentrée ou désorbitation (nullable)

TLESet                           # Jeu TLE à une époque donnée (équiv. GPSet)
- satellite Satellite .fk .nn                  # Référence au satellite concerné
- epoch datetime .nn                           # Époque TLE (UTC)
- creation_date datetime .nn                   # Date de génération Space-Track (UTC)
- element_set_no int .nn                       # ELEMENT_SET_NO (Space-Track)
- file_id int                                  # ID du fichier source (Space-Track)
- source str .max50                            # Source des données (ex: Space-Track)

TLEOrbitalElement                # Éléments orbitaux extraits du TLE
- tleset TLESet .fk .nn                         # Référence au TLESet concerné
- mean_motion float .nn                         # Révolutions par jour
- eccentricity float .nn                        # Excentricité (0..1)
- inclination float .nn                         # Inclinaison en degrés
- ra_of_asc_node float .nn                      # RAAN en degrés
- arg_of_pericenter float .nn                   # Argument du périgée en degrés
- mean_anomaly float .nn                        # Anomalie moyenne en degrés
- ephemeris_type int                            # Type d’éphéméride (0 = SGP4)
- mean_motion_dot float                         # Variation 1ère du mouvement moyen
- mean_motion_ddot float                        # Variation 2nde du mouvement moyen
- bstar float                                   # Coefficient de freinage atmosphérique
- rev_at_epoch int                              # Révolution à l’époque
- mean_element_theory str .max10                # Théorie (ex: SGP4)

TLEDynamics                      # Paramètres dérivés des TLE
- tleset TLESet .fk .nn                         # Référence au TLESet concerné
- semimajor_axis float .nn                      # Demi-grand axe en km
- period float .nn                              # Période orbitale en minutes
- apoapsis float .nn                            # Apogée en km
- periapsis float .nn                           # Périgée en km

TLEMeta                          # Métadonnées et lignes TLE brutes
- tleset TLESet .fk .nn                         # Référence au TLESet concerné
- tle_line0 str .max100                         # Ligne 0 (nom, optionnelle)
- tle_line1 str .max100 .nn                     # Ligne 1 (identifiants + BSTAR)
- tle_line2 str .max100 .nn                     # Ligne 2 (éléments orbitaux)
- ref_frame str .max10                          # Cadre de réf. (ex: TEME)
- center_name str .max10                        # Centre (ex: EARTH)
- time_system str .max10                        # Système temps (ex: UTC)
